The url is the endpoint the function is going to attack with the specified method.

When method is PUT or POST a body is required, which is always a Map<String, dynamic>. Working with JSON and RESTFUL Services may need to set the request headers. For example, 'Content-Type': 'application/json' or an auth header will be required.

This function returns a List or Map<String,dynamic> depending on the data provided by de endpoint.

dynamic request({
  required String url,
  String method = 'GET',
  Map<String, String>? headers,
  Map<String, dynamic>? body,
}) async {
  method = method.toUpperCase();
  if (!_availableMethods.contains(method)) {
    throw _RequestException(type: 'METHOD');
  } else if ((method == 'PUT' || method == 'POST') && body == null) {
    throw _RequestException(type: 'BODY');
  }

  final http.Response response = method == 'GET'
      //GET
      ? await _availableFunctions[method](Uri.parse(url), headers: headers)
      //PUT, POST, DELETE
      : await _availableFunctions[method](Uri.parse(url),
          headers: headers, body: jsonEncode(body));

  return response.body.isEmpty
      ? 'No data sent back to user'
      : jsonDecode(response.body);
}
